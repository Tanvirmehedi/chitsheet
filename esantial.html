Setting Up Build Precess With NPM Scripts
Use If Need Final Version Of Css And Project Ready For Build Process 

//npm run all install for final build css prosses one time 
npm install npm-run-all --save-dev

// watch sass 
"watch:sass" : "node-sass sass/main.scss css/style.css -w",

//live server 
"devserver":"live-server",

//npm start for sass and server same time 
"start":"npm-run-all --parallel devserver watch:sass",

//start cli 
npm run start

//compile sass
"compile:sass": "node-sass sass/main.scss css/style.comp.css",

//autoprefix css and post css comment line 
npm install autoprefixer --save-dev
npm install postcss-cli --save-dev

//prefix css script in package.jason
"prefix:css":"postcss --use autoprefixer -b 'last 10 version' css/style.comp.css -o css/style.prefix.css",

//compress css 
"compress:css":"node-sass css/style.prefix.css css/style.css --output-style compressed",

//final build in one comment 
"build:css": "npm-run-all compile:sass prefix:css compress:css"

//cli command 
npm run build:css
